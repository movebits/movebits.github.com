ubuntu edge funding level

23 July 2013

See - Ubuntu Edge funding site

#amount.rb
#Nokogiri parser every 15 min. on project page

require 'rubygems'
require 'nokogiri'
require 'open-uri'


url = "http://www.indiegogo.com/projects/ubuntu-edge"
doc = Nokogiri::HTML(open(url))
amount = doc.css(".amount").first.content

msg = "amount: " + amount.to_s + " " + Time.now.to_s + "\n"

# Time.now - (60*60*4) minus 4 hours in seconds; rackspace server is 4 hours off

output_file = "/Users/christof/Desktop/uedge.txt"


File.open(output_file, "a") do |f| 
  f.write msg
end

#cron every 15 min.

*/15 * * * * /usr/local/bin/ruby ~/amount.rb
